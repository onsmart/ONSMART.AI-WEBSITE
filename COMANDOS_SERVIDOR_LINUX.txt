OPÇÃO 4: Usando find (mais seguro, busca recursivamente)
---------------------------------------------------------
# Remover todos os PNGs de QR Code
find . -type f -name "*qrcode*.png" -delete
find . -type f -name "*qr*.png" -delete

# Remover scripts relacionados
find scripts/ -type f \( -name "*evolution*" -o -name "*whatsapp*" -o -name "*webhook*" -o -name "*qrcode*" -o -name "*cloudflared*" \) -delete

# Remover arquivos de API
find api/ -type f -name "*evolution*" -delete

# Remover serviços
find src/services/ -type f -name "*evolution*" -o -name "*whatsapp*" -delete

# Remover componentes
find src/components/ -type f -name "*WhatsApp*" -o -name "*Evolution*" -delete
find src/components/ -type d -name "*whatsapp*" -exec rm -rf {} +

# Remover hooks
find src/hooks/ -type f -name "*whatsapp*" -delete

# Remover contextos
find src/contexts/ -type f -name "*whatsapp*" -delete

# Remover páginas
find src/pages/ -type f -name "*whatsapp*" -delete
